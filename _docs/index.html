YDoc

简介

前端工程文档生成工具

安装

npm install ydoc [-g]

使用方式

命令

cd /path/to/project/
ydoc build

详细请查看命令使用说明。

脚本

var ydoc = require("ydoc");

ydoc.build('/path/to/project', options);

Gulp

var ydoc = require("ydoc");

gulp.task('ydoc', function() {
    return gulp.src('./')
        .pipe(ydoc({
            // 配置
        }));
});

Grunt

grunt.initConfig({
    ydoc: {
        // 配置
    }
});

grunt.loadNpmTasks('ydoc');
